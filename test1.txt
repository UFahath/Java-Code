        int i = 0,j=0; int temp=0,temp1=0;
        int [] a = {8,7,3,2,1}; int k =2;
        while(i<a.length)
        {
                if (i==a.length-1)
            {
                temp=a[j];
                a[j]=a[i];
                temp1=a[i-1];
                a[i-1]=temp;
                a[i]=temp1;
            }
            if(i==k)
            {
                temp=a[j];
                a[j]=a[i];
                a[i]=temp;
                i++;
                j++;
            }
